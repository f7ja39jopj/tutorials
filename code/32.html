# show tags
git tag

# create lightweight
git tag 0.0.1

# create annotated tag
git tag -a -m "" 0.0.1

# show tagged commit
git show 0.0.1

# delete tag
git tag -d 0.0.1

# create annotated (lightweight) tag later
git tag -a -m "" 0.0.1 commit_checksum

# push one tag
git push origin 0.0.1

# push all tags
git push origin --tags
