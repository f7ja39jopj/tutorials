2. Status

1. Commands

git status
git status --short

# ?? - new file created
# A_ - new file created and staged
# AM - new file created, staged and then modified
# M_ - modified and staged
# MM - modified, staged and then modified
# _M - modified and not staged
# _D - file deleted
# D_ - file deleted and staged
# R_ - file renamed and staged
# UU - merge conflict

3. Diff

1. Commands

# for _M
git diff

# for M_
git diff --staged

4. Commit

1. Commands

git commit -m "Commit message"

git commit -a -m "Commit message"

5. Remove

1. Commands

# removed and staged
git rm file_name

# removed and staged, also untracked
git rm --cached file_name

# removed and staged (modified files)
git rm -f file_name

1. Commands

8. Undoing things

1. Commands

# change commit message or add changes to the last commit
git commit --amend
Warning
Do not do this after git push
# unstage changes
git reset HEAD file_name

# remove all changes from working directory
git checkout -- file_name
Warning
Not recoverable operation

13. Tag

1. Commands

# show tags
git tag

# create lightweight
git tag 0.0.1

# create annotated tag
git tag -a -m "" 0.0.1

# show tagged commit
git show 0.0.1

# delete tag
git tag -d 0.0.1

# create annotated (lightweight) tag later
git tag -a -m "" 0.0.1 commit_checksum

# push one tag
git push origin 0.0.1

# push all tags
git push origin --tags

15. Branch

1. Commands

# show branches
git branch

# create new branch
git branch new_branch

# switch branch
git checkout new_branch

# craete new branch and switch to it
git checkout -b new_branch

# show text tree
git log --oneline --decorate --graph --all

# show last commit on each branch
git branch -v

# show last commit on each branch and track
git branch -vv

# delete remote branch
git push origin --delete branch

16. Merge

1. Commands

git checkout master
git merge new_branch     # Merge new_branch on master
git branch -d new_branch # Delete branch

# In case of merge conflict after...
git checkout master
git merge new_branch
# ... resolve conflicts
git add
git commit -m "Merge."

git branch --merged    # show merged branches
git branch --no-merged # show not merged branches

git checkout -b new_branch origin/master # track origin/master on new_branch
git branch --track origin/master         # create master branch and track it
git branch -u origin/master              # set track on existing branch

17. Rebase

1. Commands

# rebase
git checkout new_branch
git rebase master

# merge
git checkout master
git merge new_branch

# delete merged branch
git branch -d new_branch

git rebase --onto master server client

git checkout master
git merge client

git rebase master server

git checkout master
git merge server

git branch -d client
git branch -d server

18. Reset

1. Commands

# Staging area: c.txt.
# Working directory: all files.
# Untracked area: empty.
git reset --soft HEAD~

# Staging area: empty.
# Working directory: all files.
# Untracked area: c.txt.
git reset --mixed HEAD~

# Staging area: empty.
# Working directory: a.txt, b.txt.
# Untracked area: empty.
git reset --mixed HEAD~

# Staging area: c.txt, b.txt.
# Working directory: all files.
# Untracked area: empty.
git reset --soft HEAD~2

# Staging area: empty.
# Working directory: all files.
# Untracked area: c.txt, b.txt.
git reset --mixed HEAD~

# Staging area: empty.
# Working directory: a.txt.
# Untracked area: empty.
git reset --mixed HEAD~

# Restore after reset.
# Show changes list.
git reflog
git reset HEAD@{N}

# Show changes list on selected branch.
git reflog show master

19. Revert

1. Commands

# Revert the very last commit.
git revert HEAD
